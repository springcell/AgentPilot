const parseTool = require("./llm/parser")
const registry = require("./tools/registry")
const guard = require("./security/promptGuard")

module.exports = async function(req,res){

 const { llm_output } = req.body

 if(guard(llm_output)){
  return res.json({error:"prompt injection blocked"})
 }

 const tool = parseTool(llm_output)

 if(!tool){
  return res.json({error:"no tool"})
 }

 const handler = registry[tool.name]

 if(!handler){
  return res.json({error:"unknown tool"})
 }

 try{

  const result = await handler(tool)

  res.json({ok:true,result})

 }catch(e){

  res.json({ok:false,error:e.message})

 }

}
五、LLM解析器

llm/parser.js

module.exports = function(text){

 const match = text.match(/<tool>([\s\S]*?)<\/tool>/)

 if(!match) return null

 const block = match[1]

 const name = block.match(/name:\s*(.*)/)

 const args = block.match(/args:\s*(.*)/)

 return {

  name: name ? name[1].trim() : null,
  args: args ? JSON.parse(args[1]) : {}

 }

}
六、安全模块

security/promptGuard.js

module.exports = function(text){

 const bad = [

  "ignore previous",
  "bypass",
  "delete system32",
  "format disk"

 ]

 const lower = text.toLowerCase()

 return bad.some(v => lower.includes(v))

}

security/commandGuard.js

module.exports = function(cmd){

 const blocked = [

  "del ",
  "rd ",
  "format ",
  "shutdown",
  "reg ",
  "curl",
  "wget",
  "&&",
  "|",
  ">"
 ]

 const lower = cmd.toLowerCase()

 return blocked.some(v => lower.includes(v))

}

十一、命令执行器

executor/commandExecutor.js

const { exec } = require("child_process")
const guard = require("../security/commandGuard")

module.exports = function(cmd){

 return new Promise((resolve,reject)=>{

  if(guard(cmd)){
   return reject(new Error("blocked command"))
  }

  exec(cmd,(err,stdout,stderr)=>{

   if(err) reject(err)
   else resolve(stdout)

  })

 })

}

十三、LLM Prompt
你是本地AI代理。

当用户请求执行系统操作时，
只输出 tool block。

格式：

<tool>
name: open_notepad
</tool>

或

<tool>
name: list_dir
args: {"path":"C:\\Users"}
</tool>

不要解释。
不要输出其他内容。
